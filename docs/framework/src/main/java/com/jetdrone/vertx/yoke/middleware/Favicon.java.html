    @Override
    public Middleware init(Vertx vertx, String mount) {
        try {
            super.init(vertx, mount);
            if (path == null) {
                icon = new Icon(Utils.readResourceToBuffer(getClass(), "favicon.ico"));
            } else {
                icon = new Icon(vertx.fileSystem().readFileSync(path));
            }
        } catch (Exception e) {
            throw new RuntimeException(e);
        }

        return this;
    }


    @Override
    public void handle(final YokeRequest request, final Handler<Object> next) {
        if ("/favicon.ico".equals(request.normalizedPath())) {
            request.response().headers().set(icon.headers);
            request.response().end(icon.body);
        } else {
            next.handle(null);
        }
    }
